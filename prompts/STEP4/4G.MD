You are a data analysis agent for a financial crime detection hackathon.
All feature files 4A through 4F are saved in Mule-data/features/

## YOUR TASK — MERGE ALL FEATURES INTO FINAL TABLE

import pandas as pd
import numpy as np
import os

## LOAD ALL FEATURE FILES
feat_4A = pd.read_csv('Mule-data/features/feat_4A.csv',
          dtype={'account_id': str})
feat_4B = pd.read_csv('Mule-data/features/feat_4B.csv',
          dtype={'account_id': str})
feat_4C = pd.read_csv('Mule-data/features/feat_4C.csv',
          dtype={'account_id': str})
feat_4D = pd.read_csv('Mule-data/features/feat_4D.csv',
          dtype={'account_id': str})
feat_4E = pd.read_csv('Mule-data/features/feat_4E.csv',
          dtype={'account_id': str})
feat_4F = pd.read_csv('Mule-data/features/feat_4F.csv',
          dtype={'account_id': str})

## MERGE
# Start with 4A as base (has all 40,038 accounts)
features = feat_4A.copy()
drop_col = 'is_mule'

for df, name in [(feat_4B, '4B'), (feat_4C, '4C'), (feat_4D, '4D'),
                 (feat_4E, '4E'), (feat_4F, '4F')]:
    cols_to_add = [c for c in df.columns
                   if c not in ['account_id', 'is_mule']]
    features = features.merge(
        df[['account_id'] + cols_to_add],
        on='account_id', how='left')
    print(f"After merging {name}: {features.shape}")

print(f"\nFinal shape: {features.shape}")
print(f"Total features: {features.shape[1] - 2}")  # minus account_id, is_mule

## VALIDATION CHECKS

# 1. Row count
assert len(features) == 40038, f"Expected 40038 rows, got {len(features)}"
print("✓ Row count correct: 40,038")

# 2. Duplicate account_ids
assert features['account_id'].nunique() == 40038
print("✓ No duplicate account_ids")

# 3. Null analysis
null_summary = features.isnull().sum()
null_pct = (null_summary / len(features) * 100).round(2)
high_null = null_pct[null_pct > 5]
if len(high_null) > 0:
    print(f"⚠️ HIGH NULL COLUMNS (>5%):\n{high_null}")
else:
    print("✓ No columns with >5% nulls")

# 4. Infinite values
inf_cols = [c for c in features.select_dtypes(include=np.number).columns
            if np.isinf(features[c]).any()]
if inf_cols:
    print(f"⚠️ INFINITE VALUES in: {inf_cols}")
    for c in inf_cols:
        features[c] = features[c].replace([np.inf, -np.inf], np.nan)
        features[c] = features[c].fillna(features[c].median())
    print("  Fixed: replaced inf with median")
else:
    print("✓ No infinite values")

# 5. Feature correlation with is_mule
train = features[features['is_mule'].notna()].copy()
corr = train.drop(columns=['account_id']).corr()['is_mule'].abs()
corr = corr.drop('is_mule').sort_values(ascending=False)
print("\nTop 20 features by correlation with is_mule:")
print(corr.head(20))

# 6. Confirm train/test split
train_count = features['is_mule'].notna().sum()
test_count  = features['is_mule'].isna().sum()
print(f"\n✓ Train accounts: {train_count}")
print(f"✓ Test accounts:  {test_count}")

## SAVE FINAL FEATURE TABLE
features.to_csv('Mule-data/features.csv', index=False)
print("\nSaved: Mule-data/features.csv")

## SAVE FEATURE LIST FOR REFERENCE
feature_cols = [c for c in features.columns
                if c not in ['account_id', 'is_mule']]
with open('reports/FEATURE_LIST.md', 'w') as f:
    f.write("# Feature List\n\n")
    f.write(f"Total features: {len(feature_cols)}\n\n")
    for i, col in enumerate(feature_cols, 1):
        corr_val = corr.get(col, 0)
        f.write(f"{i}. `{col}` — correlation: {corr_val:.4f}\n")

print("Saved: reports/FEATURE_LIST.md")

print("\n=== STEP 4 COMPLETE ===")
print(f"Final feature table: {features.shape}")
print(f"Top feature: {corr.index[0]} (r={corr.iloc[0]:.4f})")
print("Ready for Step 5: Model Building")
