You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np

master = pd.read_csv('Mule-data/master.csv',
         dtype={'account_id': str, 'customer_id': str})
transactions = pd.read_csv('Mule-data/transactions_full.csv',
               dtype={'account_id': str, 'transaction_id': str,
               'counterparty_id': str},
               parse_dates=['transaction_timestamp'])

all_accounts = master[['account_id', 'is_mule']].copy()

transactions['hour']       = transactions['transaction_timestamp'].dt.hour
transactions['dayofweek']  = transactions['transaction_timestamp'].dt.dayofweek
transactions['day']        = transactions['transaction_timestamp'].dt.day
transactions['month']      = transactions['transaction_timestamp'].dt.month
transactions['year_month'] = transactions['transaction_timestamp'].dt.to_period('M')

## YOUR TASK — TEMPORAL & SALARY CYCLE FEATURES

### FEATURE 1 — TIME OF DAY PATTERNS
transactions['is_night'] = (
    (transactions['hour'] >= 22) | (transactions['hour'] <= 6)
).astype(int)
transactions['is_business_hours'] = (
    transactions['hour'].between(9, 17)
).astype(int)

time_feat = transactions.groupby('account_id').agg(
    night_txn_pct    = ('is_night',          'mean'),
    business_hr_pct  = ('is_business_hours', 'mean'),
).reset_index()

### FEATURE 2 — WEEKEND PATTERNS
transactions['is_weekend'] = (
    transactions['dayofweek'] >= 5
).astype(int)
weekend_feat = transactions.groupby('account_id').agg(
    weekend_txn_pct = ('is_weekend', 'mean'),
).reset_index()

### FEATURE 3 — MONTHLY VOLATILITY
# Coefficient of variation of monthly transaction counts
monthly_counts = transactions.groupby(
    ['account_id', 'year_month']).size().reset_index()
monthly_counts.columns = ['account_id', 'year_month', 'monthly_count']

monthly_stats = monthly_counts.groupby('account_id').agg(
    monthly_txn_mean   = ('monthly_count', 'mean'),
    monthly_txn_std    = ('monthly_count', 'std'),
    monthly_txn_max    = ('monthly_count', 'max'),
    active_months      = ('monthly_count', 'count'),
).reset_index()
monthly_stats['monthly_cv'] = (
    monthly_stats['monthly_txn_std'] /
    monthly_stats['monthly_txn_mean'].replace(0, np.nan)
)

### FEATURE 4 — SALARY CYCLE EXPLOITATION
# Credits on days 1–5 (salary arrival)
# Debits on days 6–15 (rapid outflow after salary)
credits = transactions[transactions['txn_type'] == 'C']
debits  = transactions[transactions['txn_type'] == 'D']

salary_credit = credits.copy()
salary_credit['is_salary_timing'] = (
    salary_credit['day'].between(1, 5)
).astype(int)
salary_feat = salary_credit.groupby('account_id').agg(
    salary_timing_credit_pct = ('is_salary_timing', 'mean'),
    total_credits_for_salary = ('transaction_id', 'count'),
).reset_index()

post_salary_debit = debits.copy()
post_salary_debit['is_post_salary'] = (
    post_salary_debit['day'].between(6, 15)
).astype(int)
post_salary_feat = post_salary_debit.groupby('account_id').agg(
    post_salary_debit_pct = ('is_post_salary', 'mean'),
).reset_index()

salary_feat = salary_feat.merge(post_salary_feat,
              on='account_id', how='left')
# Combined salary cycle score
salary_feat['salary_cycle_score'] = (
    salary_feat['salary_timing_credit_pct'] *
    salary_feat['post_salary_debit_pct'].fillna(0)
)

### FEATURE 5 — RECENCY
REFERENCE_DATE = pd.Timestamp('2025-06-30')
last_txn = transactions.groupby('account_id')[
    'transaction_timestamp'].max().reset_index()
last_txn.columns = ['account_id', 'last_txn_date']
last_txn['days_since_last_txn'] = (
    REFERENCE_DATE - last_txn['last_txn_date']
).dt.days

### MERGE ALL
feat = all_accounts.copy()
feat = feat.merge(time_feat,     on='account_id', how='left')
feat = feat.merge(weekend_feat,  on='account_id', how='left')
feat = feat.merge(monthly_stats, on='account_id', how='left')
feat = feat.merge(salary_feat.drop(
       columns=['total_credits_for_salary']),
       on='account_id', how='left')
feat = feat.merge(last_txn[['account_id', 'days_since_last_txn']],
       on='account_id', how='left')
feat = feat.fillna(0)

## VALIDATE
print("Feature shape:", feat.shape)
print("\nMule vs Legit means (temporal features):")
key = ['night_txn_pct', 'weekend_txn_pct', 'monthly_cv',
       'salary_cycle_score', 'days_since_last_txn']
print(feat[feat['is_mule'].notna()].groupby('is_mule')[key].mean())

## SAVE
feat.to_csv('Mule-data/features/feat_4F.csv', index=False)
print("\n=== TASK 4F COMPLETE — saved feat_4F.csv ===")
print(f"Shape: {feat.shape}")
