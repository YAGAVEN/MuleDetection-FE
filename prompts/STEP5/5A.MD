You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np
import os
from sklearn.model_selection import StratifiedKFold
import warnings
warnings.filterwarnings('ignore')

os.makedirs('Mule-data/models', exist_ok=True)
os.makedirs('reports/step5_model_plots', exist_ok=True)

features = pd.read_csv('Mule-data/features.csv',
           dtype={'account_id': str})

## YOUR TASK — PREPARE DATA FOR MODELING

### STEP 1 — SEPARATE TRAIN AND TEST
train = features[features['is_mule'].notna()].copy()
test  = features[features['is_mule'].isna()].copy()

print(f"Train: {len(train)} | Test: {len(test)}")
print(f"Mules: {train['is_mule'].sum()} | Legit: {(train['is_mule']==0).sum()}")
print(f"Mule rate: {train['is_mule'].mean()*100:.2f}%")

### STEP 2 — DROP WEAK AND REDUNDANT FEATURES
# Drop features with correlation < 0.02 (noise)
# Drop redundant features (keep _pct over _count where both exist)
# Drop identical features

drop_features = [
    # Redundant (credit_counterparties = unique_counterparties)
    'credit_counterparties',
    # Salary cycle (near zero correlation)
    'salary_cycle_score', 'salary_timing_credit_pct',
    'post_salary_debit_pct',
    # Digital/product (near zero)
    'digital_score', 'balance_volatility', 'balance_consistency',
    'is_overdraft', 'is_kfamily', 'is_savings',
    # Raw counts where pct version exists and is stronger
    'round_1k_count', 'round_500_count',
    'ch_upi_credit_count', 'ch_upi_debit_count',
    'amt_exact_1k', 'amt_exact_5k', 'amt_exact_10k',
    'amt_exact_100k',
    # Very weak
    'txn_velocity', 'credit_count_for_pt',
    'txn_count_post_burst_30d', 'monthly_txn_mean',
]

# Only drop if they exist
drop_features = [f for f in drop_features if f in features.columns]
print(f"\nDropping {len(drop_features)} weak/redundant features")

### STEP 3 — BUILD FEATURE MATRIX
FEATURE_COLS = [c for c in train.columns
                if c not in ['account_id', 'is_mule'] + drop_features]

print(f"Final feature count: {len(FEATURE_COLS)}")
print("\nFeatures being used:")
for i, f in enumerate(FEATURE_COLS, 1):
    print(f"  {i:3d}. {f}")

X = train[FEATURE_COLS].copy()
y = train['is_mule'].astype(int).copy()
X_test = test[FEATURE_COLS].copy()

### STEP 4 — HANDLE REMAINING NULLS
print(f"\nNull counts in X:")
null_cols = X.isnull().sum()
null_cols = null_cols[null_cols > 0]
print(null_cols if len(null_cols) > 0 else "None — clean!")

# Fill any remaining nulls with median
for col in X.columns:
    if X[col].isnull().any():
        median_val = X[col].median()
        X[col]      = X[col].fillna(median_val)
        X_test[col] = X_test[col].fillna(median_val)
        print(f"  Filled {col} nulls with median: {median_val:.4f}")

### STEP 5 — HANDLE INFINITE VALUES
for col in X.columns:
    if np.isinf(X[col]).any():
        cap = X[col].replace([np.inf, -np.inf], np.nan).quantile(0.99)
        X[col]      = X[col].replace([np.inf, -np.inf], cap)
        X_test[col] = X_test[col].replace([np.inf, -np.inf], cap)
        print(f"  Capped inf in {col} at 99th pct: {cap:.4f}")

### STEP 6 — CLASS IMBALANCE SETUP
n_legit = (y == 0).sum()
n_mule  = (y == 1).sum()
scale_pos_weight = n_legit / n_mule
print(f"\nClass imbalance ratio: {scale_pos_weight:.1f}")
print(f"(Will use as scale_pos_weight in LightGBM)")

### STEP 7 — CROSS VALIDATION SETUP
# Stratified 5-fold — preserves 1.09% mule rate in each fold
SKF = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
folds = list(SKF.split(X, y))
print(f"\nCross-validation: 5-fold stratified")
for i, (tr_idx, val_idx) in enumerate(folds):
    fold_mule_rate = y.iloc[val_idx].mean() * 100
    print(f"  Fold {i+1}: {len(val_idx)} val samples | "
          f"mule rate: {fold_mule_rate:.2f}%")

### STEP 8 — SAVE PREPARED DATA
import pickle

prep = {
    'X': X,
    'y': y,
    'X_test': X_test,
    'test_account_ids': test['account_id'].values,
    'train_account_ids': train['account_id'].values,
    'feature_cols': FEATURE_COLS,
    'folds': folds,
    'scale_pos_weight': scale_pos_weight,
}

with open('Mule-data/models/prep_data.pkl', 'wb') as f:
    pickle.dump(prep, f)

print("\nSaved: Mule-data/models/prep_data.pkl")
print("\n=== TASK 5A COMPLETE ===")
print(f"X shape: {X.shape}")
print(f"X_test shape: {X_test.shape}")
print(f"Features: {len(FEATURE_COLS)}")
print(f"Scale pos weight: {scale_pos_weight:.1f}")
