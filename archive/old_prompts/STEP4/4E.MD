You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np

master = pd.read_csv('Mule-data/master.csv',
         dtype={'account_id': str, 'customer_id': str},
         parse_dates=['account_opening_date', 'last_mobile_update_date',
                      'mule_flag_date'])
transactions = pd.read_csv('Mule-data/transactions_full.csv',
               dtype={'account_id': str, 'transaction_id': str,
               'counterparty_id': str},
               parse_dates=['transaction_timestamp'])

all_accounts = master[['account_id', 'is_mule',
                        'account_opening_date',
                        'last_mobile_update_date']].copy()

transactions = transactions.sort_values(
    ['account_id', 'transaction_timestamp'])

## YOUR TASK — PASS-THROUGH & DORMANCY FEATURES

### FEATURE 1 — RAPID PASS-THROUGH
# For each credit, find next debit and compute time gap
credits = transactions[transactions['txn_type'] == 'C'][
    ['account_id', 'transaction_timestamp']].copy()
debits  = transactions[transactions['txn_type'] == 'D'][
    ['account_id', 'transaction_timestamp']].copy()

credits.columns = ['account_id', 'credit_time']
debits.columns  = ['account_id', 'debit_time']

# Merge on account_id then find next debit after each credit
merged = pd.merge_asof(
    credits.sort_values('credit_time'),
    debits.sort_values('debit_time'),
    left_on='credit_time',
    right_on='debit_time',
    by='account_id',
    direction='forward'
)
merged['gap_hours'] = (
    merged['debit_time'] - merged['credit_time']
).dt.total_seconds() / 3600

merged = merged.dropna(subset=['gap_hours'])
merged['within_24h'] = (merged['gap_hours'] <= 24).astype(int)
merged['within_6h']  = (merged['gap_hours'] <= 6).astype(int)

passthrough = merged.groupby('account_id').agg(
    mean_passthrough_hours  = ('gap_hours',    'mean'),
    min_passthrough_hours   = ('gap_hours',    'min'),
    pct_within_24h          = ('within_24h',   'mean'),
    pct_within_6h           = ('within_6h',    'mean'),
    credit_count_for_pt     = ('credit_time',  'count'),
).reset_index()

### FEATURE 2 — DORMANCY DETECTION
# Compute gaps between consecutive transactions per account
transactions['prev_time'] = transactions.groupby('account_id')[
    'transaction_timestamp'].shift(1)
transactions['gap_days'] = (
    transactions['transaction_timestamp'] - transactions['prev_time']
).dt.days

dormancy = transactions.groupby('account_id').agg(
    max_gap_days  = ('gap_days', 'max'),
    mean_gap_days = ('gap_days', 'mean'),
).reset_index()

# Burst after longest gap
# Find the timestamp of the max gap per account
max_gap_idx = transactions.groupby('account_id')['gap_days'].idxmax()
max_gap_rows = transactions.loc[max_gap_idx.dropna().astype(int),
               ['account_id', 'transaction_timestamp']].copy()
max_gap_rows.columns = ['account_id', 'burst_start']

# Count transactions in 30 days after burst start
burst = transactions.merge(max_gap_rows, on='account_id', how='left')
burst['days_after_burst'] = (
    burst['transaction_timestamp'] - burst['burst_start']
).dt.days
burst_counts = burst[burst['days_after_burst'].between(0, 30)].groupby(
    'account_id').size().reset_index()
burst_counts.columns = ['account_id', 'txn_count_post_burst_30d']

dormancy = dormancy.merge(burst_counts, on='account_id', how='left')
dormancy['txn_count_post_burst_30d'] = (
    dormancy['txn_count_post_burst_30d'].fillna(0)
)

### FEATURE 3 — POST MOBILE CHANGE SPIKE
mobile_accounts = all_accounts[
    all_accounts['last_mobile_update_date'].notna()].copy()

def mobile_spike(group):
    acc_id = group['account_id'].iloc[0]
    mob_date = all_accounts.loc[
        all_accounts['account_id'] == acc_id,
        'last_mobile_update_date'].iloc[0]
    before = group[
        group['transaction_timestamp'] < mob_date
    ].tail(30)  # last 30 txns before
    after  = group[
        group['transaction_timestamp'] >= mob_date
    ].head(30)  # first 30 txns after
    return pd.Series({
        'txn_count_pre_mobile':  len(before),
        'txn_count_post_mobile': len(after),
    })

# Efficient approach using 30-day windows
txn_with_mobile = transactions.merge(
    all_accounts[['account_id', 'last_mobile_update_date']],
    on='account_id', how='inner')
txn_with_mobile = txn_with_mobile[
    txn_with_mobile['last_mobile_update_date'].notna()]

txn_with_mobile['days_from_mobile'] = (
    txn_with_mobile['transaction_timestamp'] -
    txn_with_mobile['last_mobile_update_date']
).dt.days

pre_mobile  = txn_with_mobile[
    txn_with_mobile['days_from_mobile'].between(-30, 0)
].groupby('account_id').size().reset_index()
pre_mobile.columns = ['account_id', 'txn_pre_mobile_30d']

post_mobile = txn_with_mobile[
    txn_with_mobile['days_from_mobile'].between(1, 30)
].groupby('account_id').size().reset_index()
post_mobile.columns = ['account_id', 'txn_post_mobile_30d']

mobile_feat = all_accounts[['account_id']].merge(
    pre_mobile,  on='account_id', how='left')
mobile_feat = mobile_feat.merge(
    post_mobile, on='account_id', how='left')
mobile_feat = mobile_feat.fillna(0)
mobile_feat['mobile_spike_ratio'] = (
    mobile_feat['txn_post_mobile_30d'] /
    mobile_feat['txn_pre_mobile_30d'].replace(0, 1)
)
mobile_feat['has_mobile_spike'] = (
    (mobile_feat['mobile_spike_ratio'] > 2) &
    (all_accounts['last_mobile_update_date'].notna().values)
).astype(int)

### MERGE ALL
feat = all_accounts[['account_id', 'is_mule']].copy()
feat = feat.merge(passthrough, on='account_id', how='left')
feat = feat.merge(dormancy,    on='account_id', how='left')
feat = feat.merge(mobile_feat.drop(columns=[]), on='account_id', how='left')
feat = feat.fillna(0)

## VALIDATE
print("Feature shape:", feat.shape)
print("\nMule vs Legit means (pass-through & dormancy):")
key = ['mean_passthrough_hours', 'pct_within_24h',
       'max_gap_days', 'txn_count_post_burst_30d',
       'mobile_spike_ratio', 'has_mobile_spike']
print(feat[feat['is_mule'].notna()].groupby('is_mule')[key].mean())

## SAVE
feat.to_csv('Mule-data/features/feat_4E.csv', index=False)
print("\n=== TASK 4E COMPLETE — saved feat_4E.csv ===")
print(f"Shape: {feat.shape}")
