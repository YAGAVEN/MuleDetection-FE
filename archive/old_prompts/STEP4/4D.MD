You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np

master = pd.read_csv('Mule-data/master.csv',
         dtype={'account_id': str, 'customer_id': str})
transactions = pd.read_csv('Mule-data/transactions_full.csv',
               dtype={'account_id': str, 'transaction_id': str,
               'counterparty_id': str},
               parse_dates=['transaction_timestamp'])

all_accounts = master[['account_id', 'is_mule']].copy()

## YOUR TASK — CHANNEL FEATURES
## EDA showed: ATW = 2.6x, NTD = 2.3x, CHQ = 2.1x, FTD = 2.0x in mules

### FEATURE 1 — HIGH SIGNAL CHANNEL PERCENTAGES
# Focus on the 4 over-represented channels from EDA
priority_channels = ['ATW', 'NTD', 'CHQ', 'FTD']

total_txn = transactions.groupby('account_id').size().reset_index()
total_txn.columns = ['account_id', 'total_count']

feat = all_accounts.copy()
feat = feat.merge(total_txn, on='account_id', how='left')

for ch in priority_channels:
    ch_count = transactions[transactions['channel'] == ch].groupby(
        'account_id').size().reset_index()
    ch_count.columns = ['account_id', f'ch_{ch.lower()}_count']
    feat = feat.merge(ch_count, on='account_id', how='left')
    feat[f'ch_{ch.lower()}_count'] = feat[f'ch_{ch.lower()}_count'].fillna(0)
    feat[f'ch_{ch.lower()}_pct'] = (
        feat[f'ch_{ch.lower()}_count'] /
        feat['total_count'].replace(0, np.nan)
    )

### FEATURE 2 — UPI CHANNELS (credit & debit separately)
upi_credit = transactions[transactions['channel'] == 'UPC']
upi_debit  = transactions[transactions['channel'] == 'UPD']

upi_c = upi_credit.groupby('account_id').size().reset_index()
upi_c.columns = ['account_id', 'ch_upi_credit_count']
upi_d = upi_debit.groupby('account_id').size().reset_index()
upi_d.columns = ['account_id', 'ch_upi_debit_count']

feat = feat.merge(upi_c, on='account_id', how='left')
feat = feat.merge(upi_d, on='account_id', how='left')
feat['ch_upi_credit_count'] = feat['ch_upi_credit_count'].fillna(0)
feat['ch_upi_debit_count']  = feat['ch_upi_debit_count'].fillna(0)
feat['ch_upi_credit_pct'] = (
    feat['ch_upi_credit_count'] /
    feat['total_count'].replace(0, np.nan)
)
feat['ch_upi_debit_pct'] = (
    feat['ch_upi_debit_count'] /
    feat['total_count'].replace(0, np.nan)
)

### FEATURE 3 — CHANNEL DIVERSITY (entropy)
channel_dist = transactions.groupby(
    ['account_id', 'channel']).size().unstack(fill_value=0)
channel_probs = channel_dist.div(channel_dist.sum(axis=1), axis=0)
# Shannon entropy
import numpy as np
entropy = -(channel_probs * np.log(channel_probs + 1e-10)).sum(axis=1)
entropy = entropy.reset_index()
entropy.columns = ['account_id', 'channel_entropy']
feat = feat.merge(entropy, on='account_id', how='left')

### FEATURE 4 — CASH-LIKE CHANNELS (ATM + OCD + CSD)
cash_channels = ['ATW', 'OCD', 'CSD']
cash_txn = transactions[transactions['channel'].isin(cash_channels)]
cash_agg = cash_txn.groupby('account_id').size().reset_index()
cash_agg.columns = ['account_id', 'cash_channel_count']
feat = feat.merge(cash_agg, on='account_id', how='left')
feat['cash_channel_count'] = feat['cash_channel_count'].fillna(0)
feat['cash_channel_pct'] = (
    feat['cash_channel_count'] /
    feat['total_count'].replace(0, np.nan)
)

### FILL NULLS
feat = feat.fillna(0)

## VALIDATE
print("Feature shape:", feat.shape)
print("\nMule vs Legit means (channel features):")
key = ['ch_atw_pct', 'ch_ntd_pct', 'ch_chq_pct',
       'ch_ftd_pct', 'channel_entropy', 'cash_channel_pct']
print(feat[feat['is_mule'].notna()].groupby('is_mule')[key].mean())

## SAVE
feat.drop(columns=['total_count'], inplace=True)
feat.to_csv('Mule-data/features/feat_4D.csv', index=False)
print("\n=== TASK 4D COMPLETE — saved feat_4D.csv ===")
print(f"Shape: {feat.shape}")
