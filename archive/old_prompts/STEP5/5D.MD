You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np
import lightgbm as lgb
import pickle
import matplotlib.pyplot as plt
from sklearn.metrics import roc_auc_score
import warnings
warnings.filterwarnings('ignore')

with open('Mule-data/models/prep_data.pkl', 'rb') as f:
    prep = pickle.load(f)
with open('Mule-data/models/tuning_results.pkl', 'rb') as f:
    tuning = pickle.load(f)

X            = prep['X']
y            = prep['y']
X_test       = prep['X_test']
folds        = prep['folds']
FEATURE_COLS = prep['feature_cols']
scale_pos_weight = prep['scale_pos_weight']
best_params  = tuning['best_params']

## YOUR TASK — FEATURE IMPORTANCE ANALYSIS

### TRAIN WITH BEST PARAMS TO GET IMPORTANCE
best_params.update({
    'objective':        'binary',
    'metric':           'auc',
    'boosting_type':    'gbdt',
    'verbose':          -1,
    'random_state':     42,
    'scale_pos_weight': scale_pos_weight,
})

importance_gain  = np.zeros(len(FEATURE_COLS))
importance_split = np.zeros(len(FEATURE_COLS))
oof_preds = np.zeros(len(X))

for fold_idx, (tr_idx, val_idx) in enumerate(folds):
    X_tr, X_val = X.iloc[tr_idx], X.iloc[val_idx]
    y_tr, y_val = y.iloc[tr_idx], y.iloc[val_idx]

    train_data = lgb.Dataset(X_tr, label=y_tr)
    val_data   = lgb.Dataset(X_val, label=y_val,
                             reference=train_data)

    model = lgb.train(
        best_params,
        train_data,
        num_boost_round=1000,
        valid_sets=[val_data],
        callbacks=[
            lgb.early_stopping(50, verbose=False),
            lgb.log_evaluation(period=-1)
        ]
    )

    oof_preds[val_idx] = model.predict(X_val)
    importance_gain  += model.feature_importance(importance_type='gain')
    importance_split += model.feature_importance(importance_type='split')

oof_auc = roc_auc_score(y, oof_preds)
print(f"Tuned OOF AUC: {oof_auc:.4f}")

### BUILD IMPORTANCE TABLE
imp_df = pd.DataFrame({
    'feature':    FEATURE_COLS,
    'gain':       importance_gain / 5,
    'split':      importance_split / 5,
})
imp_df['gain_pct']  = imp_df['gain']  / imp_df['gain'].sum()  * 100
imp_df['split_pct'] = imp_df['split'] / imp_df['split'].sum() * 100
imp_df = imp_df.sort_values('gain', ascending=False).reset_index(drop=True)

print("\nTop 30 features by importance (gain):")
print(imp_df[['feature', 'gain_pct', 'split_pct']].head(30).to_string())

### CUMULATIVE IMPORTANCE
imp_df['cumulative_gain_pct'] = imp_df['gain_pct'].cumsum()
features_for_80pct = (imp_df['cumulative_gain_pct'] <= 80).sum()
features_for_95pct = (imp_df['cumulative_gain_pct'] <= 95).sum()
print(f"\nFeatures capturing 80% of importance: {features_for_80pct}")
print(f"Features capturing 95% of importance: {features_for_95pct}")

### ZERO IMPORTANCE FEATURES
zero_imp = imp_df[imp_df['gain'] == 0]['feature'].tolist()
print(f"\nZero importance features ({len(zero_imp)}):")
print(zero_imp)

### PLOTS
fig, axes = plt.subplots(1, 2, figsize=(20, 10))

# Top 30 by gain
top30 = imp_df.head(30)
axes[0].barh(range(30), top30['gain_pct'].values, color='steelblue')
axes[0].set_yticks(range(30))
axes[0].set_yticklabels(top30['feature'].values, fontsize=9)
axes[0].invert_yaxis()
axes[0].set_xlabel('% of Total Gain')
axes[0].set_title('Top 30 Features — Importance by Gain')
axes[0].grid(True, alpha=0.3)

# Cumulative importance
axes[1].plot(range(1, len(imp_df)+1),
             imp_df['cumulative_gain_pct'].values,
             'b-', linewidth=2)
axes[1].axhline(y=80, color='r', linestyle='--', label='80%')
axes[1].axhline(y=95, color='g', linestyle='--', label='95%')
axes[1].axvline(x=features_for_80pct, color='r', linestyle=':')
axes[1].axvline(x=features_for_95pct, color='g', linestyle=':')
axes[1].set_xlabel('Number of Features')
axes[1].set_ylabel('Cumulative Importance (%)')
axes[1].set_title('Cumulative Feature Importance')
axes[1].legend()
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig('reports/step5_model_plots/02_feature_importance.png',
            dpi=150, bbox_inches='tight')
plt.close()
print("\nSaved: 02_feature_importance.png")

### SAVE
importance_results = {
    'imp_df':          imp_df,
    'oof_preds':       oof_preds,
    'oof_auc':         oof_auc,
    'zero_imp_features': zero_imp,
    'top_features_80pct': imp_df.head(features_for_80pct)['feature'].tolist(),
}

with open('Mule-data/models/importance_results.pkl', 'wb') as f:
    pickle.dump(importance_results, f)

imp_df.to_csv('reports/FEATURE_IMPORTANCE.csv', index=False)
print("Saved: reports/FEATURE_IMPORTANCE.csv")

print("\n=== TASK 5D COMPLETE ===")
print(f"Tuned AUC: {oof_auc:.4f}")
print(f"Top feature: {imp_df.iloc[0]['feature']} "
      f"({imp_df.iloc[0]['gain_pct']:.1f}% of gain)")
