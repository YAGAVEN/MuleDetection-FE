You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np
import pickle
import warnings
warnings.filterwarnings('ignore')

with open('Mule-data/models/prep_data.pkl', 'rb') as f:
    prep = pickle.load(f)
with open('Mule-data/models/final_results.pkl', 'rb') as f:
    final = pickle.load(f)

transactions = pd.read_csv('Mule-data/transactions_full.csv',
               dtype={'account_id': str, 'transaction_id': str},
               parse_dates=['transaction_timestamp'])

test_account_ids = prep['test_account_ids']
test_preds       = final['test_preds']
threshold        = final['best_threshold']

test_df = pd.DataFrame({
    'account_id': test_account_ids,
    'is_mule_prob': test_preds,
})
test_df['predicted_mule'] = (test_df['is_mule_prob'] >= threshold).astype(int)

suspected_mules = test_df[test_df['predicted_mule'] == 1]['account_id'].tolist()
print(f"Suspected mule accounts in test set: {len(suspected_mules)}")

## YOUR TASK — FIND SUSPICIOUS ACTIVITY WINDOWS

def find_suspicious_window(account_id, transactions):
    """
    Find the time window of peak suspicious activity
    using a rolling 30-day transaction velocity window.
    Returns (start_timestamp, end_timestamp)
    """
    acc_txn = transactions[
        transactions['account_id'] == account_id
    ].sort_values('transaction_timestamp').copy()

    if len(acc_txn) < 5:
        return acc_txn['transaction_timestamp'].min(), \
               acc_txn['transaction_timestamp'].max()

    acc_txn = acc_txn.set_index('transaction_timestamp')

    # Rolling 30-day count of transactions
    acc_txn['rolling_count'] = acc_txn['amount'].rolling(
        '30D').count()

    # Rolling 30-day sum of absolute amounts
    acc_txn['rolling_volume'] = acc_txn['amount'].abs().rolling(
        '30D').sum()

    # Combined score: velocity x volume
    acc_txn['activity_score'] = (
        acc_txn['rolling_count'] *
        acc_txn['rolling_volume']
    )

    # Find peak activity point
    peak_idx = acc_txn['activity_score'].idxmax()

    # Window = 30 days before peak to 30 days after peak
    window_start = peak_idx - pd.Timedelta(days=30)
    window_end   = peak_idx + pd.Timedelta(days=30)

    # Clip to actual transaction range
    window_start = max(window_start,
                       acc_txn.index.min())
    window_end   = min(window_end,
                       acc_txn.index.max())

    return window_start, window_end

### RUN WINDOW DETECTION FOR ALL SUSPECTED MULES
print("Detecting suspicious windows...")
windows = []

for i, acc_id in enumerate(suspected_mules):
    if i % 50 == 0:
        print(f"  Processing {i}/{len(suspected_mules)}...")

    start, end = find_suspicious_window(acc_id, transactions)
    windows.append({
        'account_id':       acc_id,
        'suspicious_start': start,
        'suspicious_end':   end,
    })

windows_df = pd.DataFrame(windows)
print(f"\nWindows detected for {len(windows_df)} accounts")

# Merge back to test predictions
test_df = test_df.merge(windows_df, on='account_id', how='left')

### SUMMARY STATS ON WINDOWS
mule_windows = test_df[test_df['predicted_mule'] == 1]
window_durations = (
    mule_windows['suspicious_end'] -
    mule_windows['suspicious_start']
).dt.days

print(f"\nWindow duration stats (days):")
print(f"  Mean:   {window_durations.mean():.0f}")
print(f"  Median: {window_durations.median():.0f}")
print(f"  Min:    {window_durations.min():.0f}")
print(f"  Max:    {window_durations.max():.0f}")

print(f"\nWindow start distribution by year:")
print(mule_windows['suspicious_start'].dt.year.value_counts().sort_index())

### SAVE
with open('Mule-data/models/time_windows.pkl', 'wb') as f:
    pickle.dump(test_df, f)

print("\nSaved: Mule-data/models/time_windows.pkl")
print("\n=== TASK 5F COMPLETE ===")
print(f"Suspicious windows found: {len(windows_df)}")
