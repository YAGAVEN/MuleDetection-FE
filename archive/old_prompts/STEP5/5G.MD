You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np
import pickle
import os
import warnings
warnings.filterwarnings('ignore')

with open('Mule-data/models/final_results.pkl', 'rb') as f:
    final = pickle.load(f)
with open('Mule-data/models/time_windows.pkl', 'rb') as f:
    test_df = pickle.load(f)

os.makedirs('Mule-data/submission', exist_ok=True)

## YOUR TASK — BUILD FINAL SUBMISSION FILE

### STEP 1 — BUILD SUBMISSION
submission = test_df[['account_id', 'is_mule_prob']].copy()
submission.columns = ['account_id', 'is_mule']

# Add time windows (only for predicted mules)
submission['suspicious_start'] = np.nan
submission['suspicious_end']   = np.nan

mule_mask = test_df['predicted_mule'] == 1

submission.loc[mule_mask, 'suspicious_start'] = (
    test_df.loc[mule_mask, 'suspicious_start']
    .dt.strftime('%Y-%m-%dT%H:%M:%S')
)
submission.loc[mule_mask, 'suspicious_end'] = (
    test_df.loc[mule_mask, 'suspicious_end']
    .dt.strftime('%Y-%m-%dT%H:%M:%S')
)

### STEP 2 — VALIDATE SUBMISSION
print("Validating submission...")

# Check row count
test_accounts = pd.read_csv('Mule-data/test_accounts.csv',
               dtype={'account_id': str})
assert len(submission) == len(test_accounts), \
    f"Row count mismatch: {len(submission)} vs {len(test_accounts)}"
print(f"✓ Row count: {len(submission)}")

# Check all test accounts present
missing = set(test_accounts['account_id']) - \
          set(submission['account_id'])
assert len(missing) == 0, f"Missing accounts: {missing}"
print(f"✓ All test accounts present")

# Check is_mule is probability 0-1
assert submission['is_mule'].between(0, 1).all(), \
    "is_mule values outside 0-1 range"
print(f"✓ All probabilities in [0, 1]")

# Check no nulls in is_mule
assert submission['is_mule'].notna().all()
print(f"✓ No null probabilities")

# Check time windows only for high-probability accounts
no_window_high_prob = submission[
    (submission['is_mule'] >= 0.5) &
    (submission['suspicious_start'].isna())
]
print(f"⚠️ High-prob accounts without window: "
      f"{len(no_window_high_prob)}")

### STEP 3 — SUBMISSION STATS
print(f"\nSubmission Summary:")
print(f"  Total accounts:         {len(submission)}")
print(f"  Predicted mules (>0.5): "
      f"{(submission['is_mule'] >= 0.5).sum()}")
print(f"  Predicted mules (>0.3): "
      f"{(submission['is_mule'] >= 0.3).sum()}")
print(f"  With time windows:      "
      f"{submission['suspicious_start'].notna().sum()}")
print(f"\nScore distribution:")
print(f"  <0.1:  {(submission['is_mule'] < 0.1).sum()}")
print(f"  0.1-0.3: {submission['is_mule'].between(0.1, 0.3).sum()}")
print(f"  0.3-0.5: {submission['is_mule'].between(0.3, 0.5).sum()}")
print(f"  0.5-0.7: {submission['is_mule'].between(0.5, 0.7).sum()}")
print(f"  >0.7:  {(submission['is_mule'] > 0.7).sum()}")

### STEP 4 — SAVE
submission.to_csv('Mule-data/submission/submission.csv',
                  index=False)
print(f"\nSaved: Mule-data/submission/submission.csv")

# Preview first 10 rows
print("\nSubmission preview:")
print(submission.head(10).to_string())

print("\n=== STEP 5 COMPLETE ===")
print(f"Final OOF AUC:    {final['oof_auc']:.4f}")
print(f"Submission rows:  {len(submission)}")
print(f"Predicted mules:  {(submission['is_mule'] >= 0.5).sum()}")
print("File: Mule-data/submission/submission.csv")
print("\n🎉 READY TO SUBMIT")
