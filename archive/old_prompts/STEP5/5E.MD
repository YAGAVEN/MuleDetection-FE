You are a data analysis agent for a financial crime detection hackathon.

## SETUP
import pandas as pd
import numpy as np
import lightgbm as lgb
import pickle
import matplotlib.pyplot as plt
from sklearn.metrics import (roc_auc_score, average_precision_score,
                             precision_recall_fscore_support)
import warnings
warnings.filterwarnings('ignore')

with open('Mule-data/models/prep_data.pkl', 'rb') as f:
    prep = pickle.load(f)
with open('Mule-data/models/tuning_results.pkl', 'rb') as f:
    tuning = pickle.load(f)
with open('Mule-data/models/importance_results.pkl', 'rb') as f:
    imp_results = pickle.load(f)

X                = prep['X']
y                = prep['y']
X_test           = prep['X_test']
folds            = prep['folds']
FEATURE_COLS     = prep['feature_cols']
scale_pos_weight = prep['scale_pos_weight']
best_params      = tuning['best_params']
imp_df           = imp_results['imp_df']

## YOUR TASK — TRAIN FINAL MODEL ON TOP FEATURES

### USE TOP FEATURES ONLY (capturing 95% of importance)
top_features = imp_results['top_features_80pct']

# Add back any zero-importance dropped but critical domain features
must_keep = ['is_frozen', 'unique_counterparties',
             'structuring_40k_50k_pct', 'pct_within_6h']
for f in must_keep:
    if f not in top_features and f in FEATURE_COLS:
        top_features.append(f)

print(f"Training final model with {len(top_features)} features")

X_final      = X[top_features]
X_test_final = X_test[top_features]

### FINAL PARAMS
final_params = best_params.copy()
final_params.update({
    'objective':        'binary',
    'metric':           'auc',
    'boosting_type':    'gbdt',
    'verbose':          -1,
    'random_state':     42,
    'scale_pos_weight': scale_pos_weight,
})

### FINAL CROSS VALIDATION
oof_preds  = np.zeros(len(X_final))
test_preds = np.zeros(len(X_test_final))
fold_aucs  = []
final_models = []

print("\nFinal model CV training:\n")

for fold_idx, (tr_idx, val_idx) in enumerate(folds):
    X_tr,  X_val = X_final.iloc[tr_idx],  X_final.iloc[val_idx]
    y_tr,  y_val = y.iloc[tr_idx],        y.iloc[val_idx]

    train_data = lgb.Dataset(X_tr, label=y_tr)
    val_data   = lgb.Dataset(X_val, label=y_val,
                             reference=train_data)

    model = lgb.train(
        final_params,
        train_data,
        num_boost_round=2000,
        valid_sets=[val_data],
        callbacks=[
            lgb.early_stopping(100, verbose=False),
            lgb.log_evaluation(200)
        ]
    )

    val_pred = model.predict(X_val)
    oof_preds[val_idx] = val_pred
    test_preds += model.predict(X_test_final) / 5
    fold_auc = roc_auc_score(y_val, val_pred)
    fold_aucs.append(fold_auc)
    final_models.append(model)

    print(f"Fold {fold_idx+1}: AUC={fold_auc:.4f} | "
          f"Rounds={model.best_iteration}")

oof_auc = roc_auc_score(y, oof_preds)
oof_ap  = average_precision_score(y, oof_preds)

print(f"\n{'='*50}")
print(f"FINAL OOF AUC: {oof_auc:.4f} ± {np.std(fold_aucs):.4f}")
print(f"FINAL OOF AP:  {oof_ap:.4f}")
print(f"{'='*50}")

### BEST THRESHOLD SELECTION
print("\nFinding optimal threshold for F1:")
best_f1, best_thresh = 0, 0.5
from sklearn.metrics import f1_score
for thresh in np.arange(0.05, 0.95, 0.01):
    f1 = f1_score(y, (oof_preds >= thresh).astype(int),
                  zero_division=0)
    if f1 > best_f1:
        best_f1, best_thresh = f1, thresh

print(f"Best threshold: {best_thresh:.2f} | F1: {best_f1:.4f}")
final_preds = (oof_preds >= best_thresh).astype(int)
p, r, f, _ = precision_recall_fscore_support(
    y, final_preds, average='binary')
print(f"At threshold {best_thresh:.2f}:")
print(f"  Precision: {p:.3f}")
print(f"  Recall:    {r:.3f}")
print(f"  F1:        {f:.3f}")
print(f"  Mules flagged: {final_preds.sum()} "
      f"(true mules: {y.sum()})")

### CONFUSION MATRIX
from sklearn.metrics import confusion_matrix
cm = confusion_matrix(y, final_preds)
print(f"\nConfusion Matrix (threshold={best_thresh:.2f}):")
print(f"  True Negatives:  {cm[0][0]:5d} (legit correctly cleared)")
print(f"  False Positives: {cm[0][1]:5d} (legit wrongly flagged)")
print(f"  False Negatives: {cm[1][0]:5d} (mules missed)")
print(f"  True Positives:  {cm[1][1]:5d} (mules correctly caught)")

### SAVE FINAL MODEL
final_results = {
    'oof_preds':      oof_preds,
    'test_preds':     test_preds,
    'oof_auc':        oof_auc,
    'oof_ap':         oof_ap,
    'fold_aucs':      fold_aucs,
    'best_threshold': best_thresh,
    'final_models':   final_models,
    'top_features':   top_features,
}

with open('Mule-data/models/final_results.pkl', 'wb') as f:
    pickle.dump(final_results, f)

# Save one model for inspection
final_models[0].save_model('Mule-data/models/lgbm_fold1.txt')

print("\nSaved: Mule-data/models/final_results.pkl")
print("Saved: Mule-data/models/lgbm_fold1.txt")
print("\n=== TASK 5E COMPLETE ===")
print(f"Final OOF AUC: {oof_auc:.4f}")
